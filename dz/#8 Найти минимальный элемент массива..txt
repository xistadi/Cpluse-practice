{

int arr[SIZE];

{

for (int j = 0; j < i; j++)
{
if (arr[j]==newRanomValue)
{
alreadyThere = true;

} 
}

if (!alreadyThere)
{
arr[i] = newRanomValue;
i++;
}

}

// поиск минимального элемента массива

int minValue = arr[0];

for (int i = 1; i < SIZE; i++)
{
if (arr[i]<minValue)
{
minValue = arr[i];
}
}
